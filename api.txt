Some Ideas for the API:


GET /wallboxes
[
    {
        name: <name>,
        wallboxId: <id>,
        vehicles: [
            <vehicles>
        ]
    }
]


GET /tasks
[
    {
        vehicleId: <vehicleId>,
        fromId: <id>,
        toId: <id>
    }
]


GET /vehicles
[
    <vehicle> (see below)
]


GET /vehicle/<id>
{
    modelId: <modelId>,
    vehicleId: <vehicleId>,
    licensePlate: <licensePlate>,
    batteryLevel: <batteryLevel>,
    isCharging: <isCharging>,
    hubLocation: <hubLocation>,
    state: <CHARGING|IDLE|DRIVING|PARKED|UNAVAILABLE>,
    parkingSpot: <parkingSpot>,
}

POST /vehicle/check-in?id=<id>
{
    id: <id>
}
